01

Project Overview

Our "Drawing" Machine stretches the definition of drawing. Rather than marking a canvas, the machine identifies cups placed inside its work area, moves a dispenser head directly above each cup, and delivers precise amounts of two different liquids based on settings configured through a custom web interface.

Each physical cup carries an RFID tag. When scanned, the machine loads that cup's saved recipe — or creates a new profile if the cup is new. A camera running on a separate ESP32-CAM module streams live video and uses ArUco marker detection to locate the cup in machine coordinates.

We had significant difficulties achieving consistent wiring connections and correctly mapping stepper-motor axes — challenges that dominated most of our build time and ultimately prevented us from combining all subsystems into a single working demo.

We were able to build and validate each subsystem individually — drink dispenser, gantry, and camera — but deeply struggled with integration. Pin conflicts between the RFID SPI bus, stepper drivers, and limit switches required several rewiring passes, and axis-swap bugs in the gantry firmware took significant time to isolate and fix.

// wifi credentials

// pins
#define PUMP1_PIN D0
#define PUMP2_PIN D1
#define LED_PIN   D3
#define RFID_SS_PIN  D7
#define RFID_RST_PIN D6

AsyncWebServer server(80);
AsyncWebSocket ws("/ws");
MFRC522 rfid(RFID_SS_PIN, RFID_RST_PIN);

const int NUM_PUMPS = 2;
struct PumpConfig { const char* name; uint8_t pin; };
PumpConfig pumps[NUM_PUMPS] = {
  {"Pump 1", PUMP1_PIN},
  {"Pump 2", PUMP2_PIN}
};

// machine state
String currentCupUID = "";
bool cupScanned = false;
bool machineBusy = false;
float currentTotalOz = 6.8f;
bool startRequested = false;
int currentRecipePercent[NUM_PUMPS] = {50, 50};

// simultaneous dispense state
bool dispenseActive = false;
unsigned long dispenseStartMs = 0;
int activePumpTimesMs[NUM_PUMPS] = {0, 0};
bool pumpRunning[NUM_PUMPS] = {false, false};

// cup profiles
const int MAX_CUPS = 20;
struct CupProfile {
  bool used;  String uid;
  float totalOz;  int recipePercent[NUM_PUMPS];
};
CupProfile cupProfiles[MAX_CUPS];

// calibration — empirically measured oz-to-ms curve
const int NUM_CAL_POINTS = 8;
float calOz[NUM_CAL_POINTS] = {1, 2, 3, 4, 5, 6, 7, 8};
int   calMs[NUM_CAL_POINTS] = {20460, 34383, 51676, 71567,
                                89269, 99657, 119540, 133091};

int ouncesToMs(float oz) {
  if (oz <= 0.01f) return 0;
  if (oz <= calOz[0])
    return (int)round((oz / calOz[0]) * calMs[0]);
  for (int i = 0; i < NUM_CAL_POINTS - 1; i++) {
    if (oz >= calOz[i] && oz <= calOz[i+1]) {
      float t = (oz - calOz[i]) / (calOz[i+1] - calOz[i]);
      return (int)round(calMs[i] + t*(calMs[i+1]-calMs[i]));
    }
  }
  int last = NUM_CAL_POINTS - 1;
  float slope = (float)(calMs[last]-calMs[last-1])
              / (calOz[last]-calOz[last-1]);
  return (int)round(calMs[last] + (oz-calOz[last])*slope);
}

void beginDispense() {
  for (int i = 0; i < NUM_PUMPS; i++) {
    float oz = currentTotalOz * (currentRecipePercent[i]/100.0f);
    activePumpTimesMs[i] = ouncesToMs(oz);
  }
  dispenseStartMs = millis();
  dispenseActive = true;  machineBusy = true;
  digitalWrite(LED_PIN, HIGH);
  for (int i = 0; i < NUM_PUMPS; i++) {
    if (activePumpTimesMs[i] > 0) {
      digitalWrite(pumps[i].pin, HIGH);
      pumpRunning[i] = true;
    }
  }
}

void updateDispense() {
  if (!dispenseActive) return;
  unsigned long elapsed = millis() - dispenseStartMs;
  bool anyRunning = false;
  for (int i = 0; i < NUM_PUMPS; i++) {
    if (pumpRunning[i]) {
      if ((int)elapsed >= activePumpTimesMs[i]) {
        digitalWrite(pumps[i].pin, LOW);
        pumpRunning[i] = false;
      } else { anyRunning = true; }
    }
  }
  if (!anyRunning) {
    dispenseActive = false;  machineBusy = false;
    digitalWrite(LED_PIN, LOW);
  }
}

void setup() {
  Serial.begin(115200);
  for (int i = 0; i < MAX_CUPS; i++) cupProfiles[i].used = false;
  pinMode(LED_PIN, OUTPUT);  digitalWrite(LED_PIN, LOW);
  for (int i = 0; i < NUM_PUMPS; i++) {
    pinMode(pumps[i].pin, OUTPUT);
    digitalWrite(pumps[i].pin, LOW);
  }
  SPI.begin();  rfid.PCD_Init();
  WiFi.mode(WIFI_STA);  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  ws.onEvent(onEvent);  server.addHandler(&ws);
  server.on("/", HTTP_GET, [](AsyncWebServerRequest *req) {
    req->send_P(200, "text/html", index_html);
  });
  server.begin();
}

void loop() {
  checkRFID();
  if (startRequested) {
    startRequested = false;
    if (cupScanned && !machineBusy
        && percentTotal(currentRecipePercent) == 100)
      beginDispense();
  }
  updateDispense();
  ws.cleanupClients();
}

#include <Arduino.h>
#include <WiFi.h>
#include <AsyncTCP.h>
#include <ESPAsyncWebServer.h>
#include <ArduinoJson.h>
#include "Gantry.h"

// ── Pin definitions (XIAO ESP32S3) ──────────────────────────
const uint8_t X_STEP_PIN  = D6;   // Motor A STEP
const uint8_t X_DIR_PIN   = D4;   // Motor A DIR
const uint8_t Y_STEP_PIN  = D5;   // Motor B STEP
const uint8_t Y_DIR_PIN   = D0;   // Motor B DIR
const uint8_t X_LIMIT_PIN = D7;   // Normally-open → GND
const uint8_t Y_LIMIT_PIN = D8;

// ── Machine parameters ────────────────────────────────────
// GT2 belt · 20-tooth pulley · 200-step motor · 1/16 µstep
// steps/rev = 200×16 = 3200 ; mm/rev = 20×2 = 40 → 80 steps/mm
const float STEPS_PER_MM = 80.0f;
const float MAX_SPEED    = 1200.0f;   // steps/sec
const float ACCELERATION = 800.0f;    // steps/sec²
const long  X_MAX_STEPS  = (long)(300 * STEPS_PER_MM);
const long  Y_MAX_STEPS  = (long)(300 * STEPS_PER_MM);

Gantry gantry(X_STEP_PIN, X_DIR_PIN, Y_STEP_PIN, Y_DIR_PIN,
              MAX_SPEED, ACCELERATION);

// ── Move queue (FIFO, 32 entries) ─────────────────────────
struct MoveCmd { long x; long y; };
const int QUEUE_SIZE = 32;
MoveCmd moveQueue[QUEUE_SIZE];
int qHead = 0, qTail = 0, qCount = 0;
bool isBusy = false;

bool queuePush(long x, long y) {
  if (qCount >= QUEUE_SIZE) return false;
  moveQueue[qTail] = {x, y};
  qTail = (qTail + 1) % QUEUE_SIZE;
  qCount++;  return true;
}

bool queuePop(MoveCmd &cmd) {
  if (qCount == 0) return false;
  cmd = moveQueue[qHead];
  qHead = (qHead + 1) % QUEUE_SIZE;
  qCount--;  return true;
}

// ── WebSocket command handler ──────────────────────────────
void handleWSMessage(uint8_t *data, size_t len) {
  StaticJsonDocument<256> doc;
  data[len] = 0;
  if (deserializeJson(doc, (char*)data)) return;
  const char* cmd = doc["cmd"];
  if (!cmd) return;

  if (strcmp(cmd, "home") == 0) {
    qHead = qTail = qCount = 0;
    gantry.stop();  isBusy = false;
    gantry.startHome(X_LIMIT_PIN, Y_LIMIT_PIN, 1, 200.0f, 200);

  } else if (strcmp(cmd, "gotoMM") == 0) {
    if (!gantry.isHomed()) return;
    long xs = (long)((float)doc["x"] * STEPS_PER_MM);
    long ys = (long)((float)doc["y"] * STEPS_PER_MM);
    queuePush(xs, ys);

  } else if (strcmp(cmd, "stop") == 0) {
    qHead = qTail = qCount = 0;
    gantry.stop();  isBusy = false;
  }
}

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);
  WiFi.begin("MAKERSPACE", "12345678");
  while (WiFi.status() != WL_CONNECTED) delay(500);

  gantry.stepsPerMM = STEPS_PER_MM;
  gantry.xMaxSteps  = X_MAX_STEPS;
  gantry.yMaxSteps  = Y_MAX_STEPS;
  gantry.begin();

  pinMode(X_LIMIT_PIN, INPUT_PULLUP);
  pinMode(Y_LIMIT_PIN, INPUT_PULLUP);

  // WebSocket + server setup ...
  server.begin();
  Serial.print("IP: ");  Serial.println(WiFi.localIP());
}

void loop() {
  bool homingDone = gantry.update();

  if (homingDone) {
    ws.textAll(gantry.isHomed()
      ? "{\"event\":\"homed\"}"
      : "{\"event\":\"homeFailed\"}");
  }

  if (gantry.isHomed() && !gantry.isHoming() && !isBusy && qCount > 0) {
    MoveCmd cmd;
    if (queuePop(cmd)) { gantry.goTo(cmd.x, cmd.y); isBusy = true; }
  }

  if (isBusy && gantry.isDone()) isBusy = false;

  ws.cleanupClients();
}
camera_detector.py  ·  Python · OpenCV · ArUco · Homography 
"""
Cup position detector — manual MJPEG parser version.
Bypasses OpenCV's FFMPEG backend (flaky on some platforms for ESP32-CAM streams).

Install: pip install opencv-contrib-python numpy requests
"""

import cv2, numpy as np, time, threading, json, os, requests

CAM_STREAM_URL  = "http://192.168.0.121:81/stream"
CUP_MARKER_ID   = 0        # ArUco ID attached to cup
PRINT_EVERY_SEC = 0.5

# Calibration: 4+ reference points in machine coordinates (cm).
# Place the cup at each location and press the matching key to capture.
CAL_POINTS_CM = [
    (21.0, 35.0),
    (35.0, 35.0),
    (21.0, 44.0),
    (33.0, 44.0),
]

CALIBRATION_FILE = os.path.join(os.path.dirname(__file__), "calibration.json")
cal_homography = None       # 3x3 perspective transform matrix
cal_pts_px = [None] * len(CAL_POINTS_CM)


class MjpegReader:
    """Non-blocking MJPEG reader — always exposes the latest frame."""
    def __init__(self, url):
        self.url = url
        self.frame = None
        self.lock = threading.Lock()
        self.running = True
        threading.Thread(target=self._reader, daemon=True).start()

    def _reader(self):
        while self.running:
            try:
                with requests.get(self.url, stream=True, timeout=5) as r:
                    buf = bytes()
                    for chunk in r.iter_content(chunk_size=4096):
                        buf += chunk
                        s = buf.find(b"\xff\xd8")
                        e = buf.find(b"\xff\xd9")
                        if s != -1 and e > s:
                            img = cv2.imdecode(
                                np.frombuffer(buf[s:e+2], dtype=np.uint8),
                                cv2.IMREAD_COLOR)
                            if img is not None:
                                with self.lock:
                                    self.frame = img
                            buf = buf[e+2:]
            except Exception as ex:
                print(f"stream error: {ex}")
                time.sleep(1)

    def read(self):
        with self.lock:
            return None if self.frame is None else self.frame.copy()


aruco_dict   = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
aruco_params = cv2.aruco.DetectorParameters()
detector     = cv2.aruco.ArucoDetector(aruco_dict, aruco_params)


def pixel_to_world(px, py):
    if cal_homography is None: return None
    pts = np.array([[[px, py]]], dtype=np.float64)
    out = cv2.perspectiveTransform(pts, cal_homography)
    return float(out[0,0,0])/10.0, float(out[0,0,1])/10.0   # mm → cm


def recompute_homography():
    global cal_homography
    src = [cal_pts_px[i] for i in range(len(CAL_POINTS_CM))
           if cal_pts_px[i] is not None]
    dst = [CAL_POINTS_CM[i] for i in range(len(CAL_POINTS_CM))
           if cal_pts_px[i] is not None]
    if len(src) < 4: return f"need >=4 pts, have {len(src)}"
    src_a = np.array(src,  dtype=np.float64).reshape(-1,1,2)
    dst_a = np.array([(x*10,y*10) for x,y in dst],
                     dtype=np.float64).reshape(-1,1,2)
    H, _ = cv2.findHomography(src_a, dst_a, method=0)
    if H is None: return "fit failed"
    cal_homography = H
    errs = np.linalg.norm(
        cv2.perspectiveTransform(src_a, H).reshape(-1,2)
        - np.array([(x*10,y*10) for x,y in dst]), axis=1)
    return f"fit OK  mean={errs.mean():.1f}mm  max={errs.max():.1f}mm"


def main():
    reader = MjpegReader(CAM_STREAM_URL)
    last_cup_px = None

    while True:
        frame = reader.read()
        if frame is None: time.sleep(0.01); continue

        corners, ids, _ = detector.detectMarkers(frame)
        if ids is not None:
            for mc, mid in zip(corners, ids.flatten()):
                c = mc[0]
                cx, cy = float(c[:,0].mean()), float(c[:,1].mean())
                cv2.polylines(frame, [c.astype(int)], True, (0,255,0), 2)
                if mid == CUP_MARKER_ID:
                    last_cup_px = (cx, cy)
                    world = pixel_to_world(cx, cy)
                    label = (f"cup: ({world[0]:.1f}, {world[1]:.1f}) cm"
                             if world else "cup: uncalibrated")
                    cv2.putText(frame, label, (10,30),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255,255,0), 2)

        cv2.imshow("detector  (q=quit  1-4=capture  s=save  r=reset)", frame)
        key = cv2.waitKey(1) & 0xFF
        if key == ord('q'): break
        elif ord('1') <= key <= ord('4'):
            idx = key - ord('1')
            if last_cup_px:
                cal_pts_px[idx] = last_cup_px
                n = sum(1 for p in cal_pts_px if p)
                print(f"pt {idx+1} captured — {recompute_homography()
                      if n >= 4 else f'{n}/4 needed'}")
        elif key == ord('s'):
            json.dump({"homography": cal_homography.tolist()
                       if cal_homography is not None else None,
                       "pts": cal_pts_px},
                      open(CALIBRATION_FILE,'w'), indent=2)
            print(f"saved to {CALIBRATION_FILE}")

    reader.running = False
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()

Reflection

Each subsystem — drink dispenser, gantry controller, and camera detector — worked in isolation. The drink machine correctly reads RFID tags, saves per-cup recipes, and runs timed pump sequences. The gantry homes reliably and follows queued move commands. The camera tracks ArUco markers and maps pixel coordinates to machine space via a perspective homography.

Integration proved harder than expected. The RFID module's SPI bus conflicts with the stepper driver pins on the XIAO ESP32S3, requiring careful pin reassignment. Axis swaps between physical motors and firmware labels added several debug sessions. Ultimately we ran out of time to merge all three systems into a single demo, but each piece is documented and functional independently.
